309 template<class T>
310 __attribute__((noinline))
311 static float InterpolateTime(T* cap, size_t a, float voltage)
312 {
313 AssertTypeIsAnalogWaveform(cap);
314
315 //If the voltage isn't between the two points, abort
316 float fa = cap->m_samples[a];
317 float fb = cap->m_samples[a+1];
318 bool ag = (fa > voltage);
319 bool bg = (fb > voltage);
320 if( (ag && bg) || (!ag && !bg) )
321 return 0;
322
323 //no need to divide by time, sample spacing is normalized to 1 timebase unit
324 float slope = (fb - fa);
325 float delta = voltage - fa;
326 return delta / slope;
327 }

442 template<class T>
443 __attribute__((noinline))
444 static float GetBaseVoltage(T* cap)
445 {
446 AssertTypeIsAnalogWaveform(cap);
447
448 float vmin = GetMinVoltage(cap);
449 float vmax = GetMaxVoltage(cap);
450 float delta = vmax - vmin;
451 const int nbins = 100;
452 auto hist = MakeHistogram(cap, vmin, vmax, nbins);
453
454 //Find the highest peak in the first quarter of the histogram
455 size_t binval = 0;
456 int idx = 0;
457 for(int i=0; i<(nbins/4); i++)
458 {
459 if(hist[i] > binval)
460 {
461 binval = hist[i];
462 idx = i;
463 }
464 }
465
466 float fbin = (idx + 0.5f)/nbins;
467 return fbin*delta + vmin;
468 }

560 template<class T>
561 __attribute__((noinline))
562 static std::vector<size_t> MakeHistogram(T* cap, float low, float high, size_t bins)
563 {
564 AssertTypeIsAnalogWaveform(cap);
565
566 std::vector<size_t> ret;
567 for(size_t i=0; i<bins; i++)
568 ret.push_back(0);
569
570 //Early out if we have zero span
571 if(bins == 0)
572 return ret;
573
574 float delta = high-low;
575
576 for(float v : cap->m_samples)
577 {
578 float fbin = (v-low) / delta;
579 size_t bin = floor(fbin * bins);
580 if(fbin < 0)
581 bin = 0;
582 else
583 bin = std::min(bin, bins-1);
584 ret[bin] ++;
585 }
586
587 return ret;
588 }

601 {
602 if(s)
603 return MakeHistogram(s, low, high, bins);
604 else
605 return MakeHistogram(u, low, high, bins);
606 }

617 template<class T>
618 __attribute__((noinline))
619 static std::vector<size_t> MakeHistogramClipped(T* cap, float low, float high, size_t bins)
620 {
621 AssertTypeIsAnalogWaveform(cap);
622
623 std::vector<size_t> ret;
624 for(size_t i=0; i<bins; i++)
625 ret.push_back(0);
626
627 //Early out if we have zero span
628 if(bins == 0)
629 return ret;
630
631 float delta = high-low;
632
633 for(float v : cap->m_samples)
634 {
635 float fbin = (v-low) / delta;
636 // must cast through a signed int type to avoid UB (e.g. saturates to 0 on arm64) [conv.fpint]
637 size_t bin = static_cast<ssize_t>(floor(fbin * bins));
638 if(bin >= bins) //negative values wrap to huge positive and get caught here
639 continue;
640 ret[bin] ++;
641 }
642
643 return ret;
644 }

658 template<class T, class R, class S>
659 __attribute__((noinline))
660 static void SampleOnAnyEdges(T* data, R* clock, SparseWaveform<S>& samples)
661 {
662 //Compile-time check to make sure inputs are correct types
663 AssertTypeIsDigitalWaveform(clock);
664 AssertSampleTypesAreSame(data, &samples);
665
666 samples.clear();
667 samples.SetGpuAccessHint(AcceleratorBuffer<S>::HINT_NEVER); //assume we're being used as part of a CPU-side filter
668 samples.PrepareForCpuAccess();
669
670 //TODO: split up into blocks and multithread?
671 //TODO: AVX vcompress?
672
673 size_t len = clock->size();
674 size_t dlen = data->size();
675
676 size_t ndata = 0;
677 for(size_t i=1; i<len; i++)
678 {
679 //Throw away clock samples until we find an edge
680 if(clock->m_samples[i] == clock->m_samples[i-1])
681 continue;
682
683 //Throw away data samples until the data is synced with us
684 int64_t clkstart = GetOffsetScaled(clock, i);
685 while( (ndata+1 < dlen) && (GetOffsetScaled(data, ndata+1) < clkstart) )
686 ndata ++;
687 if(ndata >= dlen)
688 break;
689
690 //Add the new sample
691 samples.m_offsets.push_back(clkstart);
692 samples.m_samples.push_back(data->m_samples[ndata]);
693 }
694
695 //Compute sample durations
696 #ifdef __x86_64__
697 if(g_hasAvx2)
698 FillDurationsAVX2(samples);
699 else
700 #endif
701 FillDurationsGeneric(samples);
702
703 samples.MarkModifiedFromCpu();
704 }

918 samples.SetGpuAccessHint(AcceleratorBuffer<float>::HINT_NEVER); //assume we're being used as part of a CPU-side filter
919
920 //TODO: split up into blocks and multithread?
921 //TODO: AVX vcompress
922
923 size_t len = clock->size();
924 size_t dlen = data->size();
925
926 size_t ndata = 0;
927 for(size_t i=1; i<len; i++)
928 {
929 //Throw away clock samples until we find an edge
930 if(clock->m_samples[i] == clock->m_samples[i-1])
931 continue;
932
933 //Throw away data samples until the data is synced with us
934 int64_t clkstart = GetOffsetScaled(clock, i);
935 while( (ndata+1 < dlen) && (GetOffsetScaled(data, ndata+1) < clkstart) )
936 ndata ++;
937 if(ndata >= dlen)
938 break;
939
940 //Find the fractional position of the clock edge
941 int64_t tsample = GetOffsetScaled(data, ndata);
942 int64_t delta = clkstart - tsample;
943 float frac = delta * 1.0 / data->m_timescale;
944
945 //Add the new sample
946 samples.m_offsets.push_back(clkstart);
947 samples.m_samples.push_back(InterpolateValue(data, ndata, frac));
948 }
949
950 //Compute sample durations
951 #ifdef __x86_64__
952 if(g_hasAvx2)
953 FillDurationsAVX2(samples);
954 else
955 #endif
956 FillDurationsGeneric(samples);
957
958 samples.MarkModifiedFromCpu();
959 }
