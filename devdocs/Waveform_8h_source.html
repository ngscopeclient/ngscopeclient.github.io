446template<class S>
447class SparseWaveform : public SparseWaveformBase
448{
449public:
450
456 SparseWaveform(const std::string& name = "")
457 {
458 Rename(name);
459
460 //Default data to CPU/GPU mirror
461 m_samples.SetCpuAccessHint(AcceleratorBuffer<S>::HINT_LIKELY);
462 m_samples.SetGpuAccessHint(AcceleratorBuffer<S>::HINT_LIKELY);
463 m_samples.PrepareForCpuAccess();
464 }
465
466 virtual void Rename(const std::string& name = "") override
467 {
468 if(name.empty())
469 {
470 m_samples.SetName(std::string("UniformWaveform<") + typeid(S).name() + ">.m_samples");
471 m_offsets.SetName(std::string("SparseWaveform<") + typeid(S).name() + ">.m_offsets");
472 m_durations.SetName(std::string("SparseWaveform<") + typeid(S).name() + ">.m_durations");
473 }
474 else
475 {
476 m_samples.SetName(name + ".m_samples");
477 m_offsets.SetName(name + ".m_offsets");
478 m_durations.SetName(name + ".m_durations");
479 }
480 }
481
485 SparseWaveform(UniformWaveform<S>& rhs)
486 {
487 m_samples.SetName(std::string("SparseWaveform<") + typeid(S).name() + ">.m_samples");
488 m_offsets.SetName(std::string("SparseWaveform<") + typeid(S).name() + ">.m_offsets");
489 m_durations.SetName(std::string("SparseWaveform<") + typeid(S).name() + ">.m_durations");
490
491 m_samples.SetCpuAccessHint(AcceleratorBuffer<S>::HINT_LIKELY);
492 m_samples.SetGpuAccessHint(AcceleratorBuffer<S>::HINT_LIKELY);
493 m_samples.PrepareForCpuAccess();
494
495 //Copy sample data
496 Resize(rhs.size());
497 m_samples.CopyFrom(rhs.m_samples);
498
499 //Generate offset/duration values
500 //TODO: is it worth vectorizing this since this is mostly meant to be a compatibility layer?
501 for(size_t i=0; i<m_offsets.size(); i++)
502 {
503 m_offsets[i] = i;
504 m_durations[i] = 1;
505 }
506 }
507
508 virtual ~SparseWaveform()
509 {}
510
512 AcceleratorBuffer<S> m_samples;
513
514 virtual void Resize(size_t size) override
515 {
516 m_offsets.resize(size);
517 m_durations.resize(size);
518 m_samples.resize(size);
519 }
520
521 virtual size_t size() const override
522 { return m_samples.size(); }
523

526 m_offsets.clear();

536 }

542 m_samples.PrepareForGpuAccess();
543 }
