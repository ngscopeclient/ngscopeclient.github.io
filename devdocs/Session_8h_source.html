53class InstrumentConnectionState
54{
55public:
56 InstrumentConnectionState(InstrumentThreadArgs args)
57 {
58 m_shuttingDown = false;
59 args.shuttingDown = &m_shuttingDown;
60 m_thread = std::make_unique<std::thread>(InstrumentThread, args);
61 m_lastTriggerState = Oscilloscope::TRIGGER_MODE_WAIT;
62 }
63
64 ~InstrumentConnectionState()
65 { Close(); }
66
67 void Close()
68 {
69 if(m_thread)
70 {
71 //Terminate the thread
72 m_shuttingDown = true;
73 m_thread->join();
74 }
75 m_thread = nullptr;
76 }
